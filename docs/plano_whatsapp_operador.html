def _usuario_por_telefone_operador(telefone: str, db: Session) -> "Usuario | None":
    """Busca Usuario pelo telefone_operador (com fallback para Empresa.telefone_operador)."""
    dig = _digitos_telefone(telefone)
    if len(dig) < 8:
        return None
    sufixo = dig[-8:]

    # 1. Busca por Usuario.telefone_operador (multi-operador)
    if _is_postgresql(db):
        digits_expr = func.regexp_replace(
            Usuario.telefone_operador, r"[^0-9]", "", "g"
        )
        return (
            db.query(Usuario)
            .filter(
                Usuario.telefone_operador.isnot(None),
                func.right(digits_expr, 8) == sufixo,
                Usuario.ativo == True,
            )
            .first()
        )
    # fallback SQLite
    for u in db.query(Usuario).filter(Usuario.telefone_operador.isnot(None)).all():
        if _digitos_telefone(u.telefone_operador or "")[-8:] == sufixo:
            return u
    return None

"""
Camada de interações interativas para o canal WhatsApp do operador.
Abstrai Poll, Lista e Texto estruturado — sem depender de sendButtons.
"""
import logging
import httpx
from app.core.config import settings

logger = logging.getLogger(__name__)

async def enviar_poll_confirmacao(
    telefone: str,
    pergunta: str,
    opcoes: list[str],
    instancia: str | None = None,
) -> bool:
    """
    Envia enquete nativa do WhatsApp para confirmação.
    Mais estável que sendButtons na Evolution API atual.
    opcoes: ex: ["Confirmar", "Cancelar", "Editar"]
    """
    inst = instancia or settings.EVOLUTION_INSTANCE
    url  = f"{settings.EVOLUTION_API_URL.rstrip('/')}/message/sendPoll/{inst}"
    phone = _fmt_phone(telefone)
    payload = {
        "number": phone,
        "name": pergunta,
        "values": opcoes,
        "selectableCount": 1,
    }
    try:
        async with httpx.AsyncClient(timeout=10) as client:
            r = await client.post(url, json=payload,
                headers={"apikey": settings.EVOLUTION_API_KEY,
                         "Content-Type": "application/json"})
            return r.status_code in (200, 201)
    except Exception as e:
        logger.warning("[Poll] Falha: %s", e)
        return False

async def enviar_lista_selecao(
    telefone: str,
    titulo: str,
    descricao: str,
    secoes: list[dict],  # [{"titulo": str, "itens": [{"id": str, "titulo": str, "desc": str}]}]
    botao_texto: str = "Ver opções",
    instancia: str | None = None,
) -> bool:
    """Envia menu de lista (sendList) — funciona para selecionar clientes, serviços etc."""
    inst = instancia or settings.EVOLUTION_INSTANCE
    url  = f"{settings.EVOLUTION_API_URL.rstrip('/')}/message/sendList/{inst}"
    sections = [
        {"title": s["titulo"],
         "rows": [{"title": i["titulo"], "description": i.get("desc", ""),
                    "rowId": i["id"]} for i in s["itens"]]}
        for s in secoes
    ]
    payload = {
        "number": _fmt_phone(telefone),
        "title": titulo,
        "description": descricao,
        "buttonText": botao_texto,
        "sections": sections,
    }
    try:
        async with httpx.AsyncClient(timeout=10) as client:
            r = await client.post(url, json=payload,
                headers={"apikey": settings.EVOLUTION_API_KEY,
                         "Content-Type": "application/json"})
            return r.status_code in (200, 201)
    except Exception as e:
        logger.warning("[Lista] Falha: %s", e)
        return False

def texto_opcoes_numeradas(titulo: str, descricao: str, opcoes: list[str]) -> str:
    """Fallback universal: texto formatado com opções numeradas."""
    emojis = ["1️⃣", "2️⃣", "3️⃣", "4️⃣", "5️⃣"]
    linhas = [f"*{titulo}*", "", descricao, ""]
    for i, op in enumerate(opcoes):
        em = emojis[i] if i < len(emojis) else f"{i+1}."
        linhas.append(f"{em} *{op}*")
    return "\n".join(linhas)

def _fmt_phone(telefone: str) -> str:
    from app.utils.phone import normalize_phone_number
    normalized = normalize_phone_number(telefone)
    return normalized or telefone

"""
Orquestra o fluxo completo do operador via WhatsApp:
1. Chama assistente_unificado (mesmo pipeline do chat web)
2. Interpreta AIResponse e envia interação adequada (Poll, Lista ou texto)
3. Gerencia confirmações pendentes via SessionStore
"""
import logging
from sqlalchemy.orm import Session
from app.models.models import Usuario, Empresa
from app.services.cotte_ai_hub import AIResponse, assistente_unificado
from app.services.whatsapp_service import enviar_mensagem_texto
from app.services.operador_interacao_service import (
    enviar_poll_confirmacao, texto_opcoes_numeradas
)
from app.services.cotte_context_builder import SessionStore

logger = logging.getLogger(__name__)

# ── Chave extra na SessionStore para pending_action WhatsApp ────────────────

def _wpp_sessao_id(telefone: str) -> str:
    return f"wpp-op-{telefone}"

def _salvar_pending_wpp(sessao_id: str, pending: dict):
    """Guarda o pending_action e confirmation_token na sessão."""
    from app.services.cotte_context_builder import _sessions
    if sessao_id in _sessions:
        _sessions[sessao_id]["wpp_pending"] = pending

def _recuperar_pending_wpp(sessao_id: str) -> dict | None:
    from app.services.cotte_context_builder import _sessions
    return _sessions.get(sessao_id, {}).get("wpp_pending")

def _limpar_pending_wpp(sessao_id: str):
    from app.services.cotte_context_builder import _sessions
    if sessao_id in _sessions:
        _sessions[sessao_id].pop("wpp_pending", None)

# ── Detecção de confirmação por texto ───────────────────────────────────────

_CONFIRMAR = {"confirmar", "sim", "ok", "1", "s"}
_CANCELAR  = {"cancelar", "não", "nao", "2", "n"}

def _detectar_resposta_poll(mensagem: str) -> str | None:
    """Retorna 'CONFIRMAR', 'CANCELAR' ou None."""
    norm = mensagem.strip().lower()
    if norm in _CONFIRMAR: return "CONFIRMAR"
    if norm in _CANCELAR:  return "CANCELAR"
    return None

# ── Entry point principal ───────────────────────────────────────────────────

async def processar_operador_wpp(
    telefone: str,
    mensagem: str,
    operador: Usuario,
    db: Session,
    empresa: Empresa,
) -> None:
    sessao_id = _wpp_sessao_id(telefone)

    # 1. Verificar se há pending_action esperando confirmação
    pending = _recuperar_pending_wpp(sessao_id)
    if pending:
        resposta_poll = _detectar_resposta_poll(mensagem)
        if resposta_poll == "CONFIRMAR":
            _limpar_pending_wpp(sessao_id)
            # Re-chama o assistente com o confirmation_token
            ai_resp = await assistente_unificado(
                mensagem=mensagem,
                sessao_id=sessao_id,
                db=db,
                empresa_id=empresa.id,
                usuario_id=operador.id,
                permissoes=operador.permissoes or {},
                is_gestor=operador.is_gestor,
                # confirmation_token só existe no v2; passa no contexto da sessão
            )
            await _enviar_resposta(telefone, ai_resp, sessao_id, empresa, db)
            return
        elif resposta_poll == "CANCELAR":
            _limpar_pending_wpp(sessao_id)
            await enviar_mensagem_texto(telefone, "Ação cancelada.", empresa=empresa)
            return
        # Se não é confirmação/cancelamento: processa como nova mensagem normalmente
        _limpar_pending_wpp(sessao_id)

    # 2. Processar com assistente_unificado
    try:
        ai_resp = await assistente_unificado(
            mensagem=mensagem,
            sessao_id=sessao_id,
            db=db,
            empresa_id=empresa.id,
            usuario_id=operador.id,
            permissoes=operador.permissoes or {},
            is_gestor=operador.is_gestor,
        )
    except Exception as e:
        logger.error("[OperadorWPP] Erro no assistente: %s", e)
        await enviar_mensagem_texto(
            telefone, "Erro interno. Tente novamente.", empresa=empresa
        )
        return

    await _enviar_resposta(telefone, ai_resp, sessao_id, empresa, db)


async def _enviar_resposta(
    telefone: str, ai_resp: AIResponse, sessao_id: str, empresa: Empresa, db
) -> None:
    """Interpreta AIResponse e envia a mensagem/interação adequada."""

    # Caso 1: ação pendente — envia Poll de confirmação
    if ai_resp.pending_action:
        _salvar_pending_wpp(sessao_id, ai_resp.pending_action)
        dados = ai_resp.dados or {}
        numero = dados.get("numero", "")
        total  = dados.get("total", 0)
        cliente = dados.get("cliente_nome", "")
        servico = dados.get("servico", "")
        pergunta = (
            f"Criar orçamento {numero} — {cliente}\n{servico} · R$ {total:.2f}"
            if numero else
            (ai_resp.resposta or "Confirmar ação?")[:100]
        )
        ok = await enviar_poll_confirmacao(
            telefone, pergunta, ["Confirmar", "Cancelar"], instancia=None
        )
        if not ok:
            # Fallback: texto numerado
            texto = texto_opcoes_numeradas(
                "Confirmação necessária", pergunta, ["Confirmar", "Cancelar"]
            )
            await enviar_mensagem_texto(telefone, texto, empresa=empresa)
        return

    # Caso 2: orçamento criado — notificação rica
    if ai_resp.tipo_resposta == "orcamento_criado":
        dados = ai_resp.dados or {}
        numero = dados.get("numero", "?")
        total  = dados.get("total", 0)
        seq    = numero.split("-")[1] if "-" in numero else numero
        texto  = (
            f"✅ *Orçamento {numero} criado!*\n"
            f"Total: *R$ {total:.2f}*\n\n"
            f"Para enviar ao cliente: *enviar {seq}*"
        )
        await enviar_mensagem_texto(telefone, texto, empresa=empresa)
        return

    # Caso 3: resposta genérica
    texto = (ai_resp.resposta or "Sem resposta.").strip()
    if not ai_resp.sucesso:
        texto = f"❌ {texto}"
    await enviar_mensagem_texto(telefone, texto, empresa=empresa)

Quando o operador clica em uma opção do Poll, a Evolution API envia um evento MESSAGES_UPSERT com um payload diferente de texto normal. O Baileys entrega a resposta de poll como pollUpdateMessage — precisamos extraí-la.

# Adicionar extração de resposta de poll no parseamento do payload
def _extrair_texto_mensagem(data: dict) -> str | None:
    """Extrai texto de mensagem normal OU de voto em poll."""
    msg = data.get("message", {}) or {}

    # Mensagem de texto normal
    texto = (
        msg.get("conversation")
        or (msg.get("extendedTextMessage") or {}).get("text")
    )
    if texto:
        return texto

    # Resposta de poll — Evolution/Baileys encapsula assim:
    poll_upd = msg.get("pollUpdateMessage") or {}
    if poll_upd:
        votes = poll_upd.get("vote", {}).get("selectedOptions", [])
        if votes:
            return votes[0]  # ex: "Confirmar" ou "Cancelar"

    return None
Notas

  • O texto extraído do poll ("Confirmar" / "Cancelar") entra normalmente no processar_mensagem() e é detectado pelo _detectar_resposta_poll()

T4.2 Calibrar textos de resposta para o canal WhatsApp
editar
▶
Problema

O assistente_unificado gera respostas ricas com markdown (tabelas, listas longas) adequadas ao chat web. No WhatsApp, o markdown é limitado — apenas *negrito*, _itálico_ e ~tachado~.

Solução
def sanitizar_para_whatsapp(texto: str) -> str:
    """Converte markdown rico para formato WhatsApp."""
    import re
    # Remove headers markdown
    texto = re.sub(r'^#{1,6}\s+', '', texto, flags=re.MULTILINE)
    # Converte **: bold → *bold*
    texto = re.sub(r'\*\*(.+?)\*\*', r'*\1*', texto)
    # Remove tabelas (linhas com |)
    texto = re.sub(r'\|.*\|', '', texto)
    # Limita a 1000 chars (WhatsApp quebra mensagens longas)
    if len(texto) > 900:
        texto = texto[:900] + "\n_(mensagem truncada — veja detalhes no app)_"
    return texto.strip()

Chamar essa função em _enviar_resposta() antes de enviar_mensagem_texto().
